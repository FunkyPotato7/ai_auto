**4. `script.js` (JavaScript - Quizz one Functionality):**

```javascript
document.addEventListener('DOMContentLoaded', function() {
    const quizzOneLink = document.getElementById('quizz-one-link');
    const popup = document.getElementById('popup');
    const closeButton = document.querySelector('.close-button');

    if (quizzOneLink && popup && closeButton) {
        quizzOneLink.addEventListener('click', function(event) {
            event.preventDefault(); // Prevent the link from navigating

            // Close the menu (assuming it's a simple navigation)
            const nav = document.querySelector('nav'); // Or however you select your navigation
            if(nav){
                nav.style.display = 'none'; // Or perhaps nav.classList.remove('open') if you're using classes for menu states.
            }

            // Show the popup
            popup.style.display = 'block';
        });

        closeButton.addEventListener('click', function() {
            popup.style.display = 'none';
            //Re-open the menu after popup is closed.
             const nav = document.querySelector('nav');
             if(nav){
                nav.style.display = 'block';
             }
        });

        // Close the popup if the user clicks outside the popup content
        window.addEventListener('click', function(event) {
            if (event.target === popup) {
                popup.style.display = 'none';
                const nav = document.querySelector('nav');
                 if(nav){
                    nav.style.display = 'block';
                 }
            }
        });
    } else {
        console.error("One or more elements not found. Check your HTML.");
    }
});
```
