<script>
  document.addEventListener('DOMContentLoaded', function() {
    const videoPopupButtons = document.querySelectorAll('.video-popup-button');
    const closeButtons = document.querySelectorAll('.close-video-popup');

    videoPopupButtons.forEach(button => {
      button.addEventListener('click', function(event) {
        event.preventDefault();
        const videoId = this.dataset.videoId;
        const videoPopup = document.getElementById(videoId);
        if (videoPopup) {
          videoPopup.style.display = 'flex';
        }
      });
    });

    closeButtons.forEach(button => {
      button.addEventListener('click', function() {
        const videoPopup = this.closest('.video-popup');
        if (videoPopup) {
          videoPopup.style.display = 'none';
           //stop video when closing modal
          videoPopup.querySelector('iframe').src = videoPopup.querySelector('iframe').src;
        }
      });
    });
  });
</script>
